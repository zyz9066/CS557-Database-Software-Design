CREATE PROCEDURE countBook

@isbn_mean : nvarchar(20),
@value : nvarchar(20) 
AS 
BEGIN 
	select count(*) as field from `BOOK` where substring(@isbn_mean,3,1) = @ value
end

    

CREATE PROCEDURE countOder

@isbn_mean: nvarchar(20),
@value : nvarchar(20) 
As
Begin
	Select count(*) as field from `ORDER` where substring(@isbn_mean,3,1) = @value
End
